# This is the script for enabling azure hybrid benefit for a single Azure Stack HCI cluster.

# Code starts here

# Define tenant, subscription, resourceGroup and cluster
$tenant = ""
$subscription = ""
$resourceGroup = ""
$cluster = ""

# Set the active Azure subscription to the one specified in `$subscription`
Set-AzContext -Subscription $subscription

# Enable the software assurance benefit by sending an API call to Azure
Invoke-AzRestMethod -Method POST `
-SubscriptionId $subscription `
-ResourceGroupName $resourceGroup `
-ResourceProviderName "Microsoft.AzureStackHCI" `
-ResourceType "cluster" `
-Name ($cluster + "/extendSoftwareAssuranceBenefit") `
-ApiVersion "2023-02-01" `
-Payload '{"properties": { "softwareAssuranceIntent": "Enable" }}'
