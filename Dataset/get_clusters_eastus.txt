# This script gets Azure Stack HCI clusters in a given location

#Code starts here

$subscriptionId = ""
$resourceGroup = ""
$tenantId = ""
$location = ""

#Using AZ POWERSHELL

Connect-AzAccount -Tenant $tenantId
Set-AzContext

# get clusters in location EastUS 
Get-AzStackHciCluster -ResourceGroupName $resourceGroup | Where-Object location -eq $location | Select-Object Name, Id
